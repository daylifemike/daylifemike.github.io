2-way data binding
jQuery
<h1>Hello {{ name }}!</h1>
<input type="text"/>

<script>
  $(function(){
    var name = null;
    var $output = $('h1');
    var $input = $('input');
    var template = $output.text();

    function updateName(value){
      name = value;
      $output.text(template.replace('{{ name }}', value));
    }

    $input.on('keyup', function(){
      updateName($(this).val());
    });
  });
</script>
